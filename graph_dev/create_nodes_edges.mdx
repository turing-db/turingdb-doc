    ```python
    # client.set_graph("mygraph")  # set working graph is not already done
    client.checkout(my_change)
    
    # Create a node Person (Jane) - Edge (knows) - node (John)
    client.query("CREATE (n:Person {name: 'Jane'})-[e:KNOWS]->(m:Person {name: 'John'})")
    
    # Commit the change
    client.query("COMMIT")
    ```
