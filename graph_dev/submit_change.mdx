---

<Tabs>
  <Tab title="Cypher">
    ```jsx
    // Submit changes done to the database
    CHANGE SUBMIT
    
    // Checkout to main branch
    checkout
    ```
  </Tab>
  <Tab title="Python SDK">
    ```python
    # Submit changes done to the database
    client.query("CHANGE SUBMIT")
    
    # Checkout to main branch
    client.checkout()
    ```
  </Tab>
</Tabs>
